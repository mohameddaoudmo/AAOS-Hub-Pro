class CarBluetoothProfiles(private val context: Context) {

    private var a2dpProfile: BluetoothA2dp? = null
    private var hfpProfile:  BluetoothHeadset? = null

    private val btAdapter = BluetoothAdapter.getDefaultAdapter()

    // --- ??????? ?? Profile Proxy ---
    fun initProfiles() {
        // A2DP Ś ?????
        btAdapter.getProfileProxy(context, object : BluetoothProfile.ServiceListener {
            override fun onServiceConnected(profile: Int, proxy: BluetoothProfile) {
                a2dpProfile = proxy as BluetoothA2dp
            }
            override fun onServiceDisconnected(profile: Int) { a2dpProfile = null }
        }, BluetoothProfile.A2DP)

        // HFP Ś ?????????
        btAdapter.getProfileProxy(context, object : BluetoothProfile.ServiceListener {
            override fun onServiceConnected(profile: Int, proxy: BluetoothProfile) {
                hfpProfile = proxy as BluetoothHeadset
            }
            override fun onServiceDisconnected(profile: Int) { hfpProfile = null }
        }, BluetoothProfile.HEADSET)
    }

    // --- ??? ???? ??? Profile ---
    fun getA2dpState(device: BluetoothDevice): Int {
        return a2dpProfile?.getConnectionState(device)
            ?: BluetoothProfile.STATE_DISCONNECTED
    }

    // --- ?????? ???? A2DP ---
    private val a2dpReceiver = object : BroadcastReceiver() {
        override fun onReceive(ctx: Context, intent: Intent) {
            val state = intent.getIntExtra(
                BluetoothProfile.EXTRA_STATE,
                BluetoothProfile.STATE_DISCONNECTED
            )
            when (state) {
                BluetoothProfile.STATE_CONNECTED    -> Log.d("A2DP", "Audio connected")
                BluetoothProfile.STATE_DISCONNECTED -> Log.d("A2DP", "Audio disconnected")
            }
        }
    }

    fun register() {
        context.registerReceiver(a2dpReceiver,
            IntentFilter(BluetoothA2dp.ACTION_CONNECTION_STATE_CHANGED))
    }
}

class BleAdvertiser(private val context: Context) {

        val settings = AdvertiseSettings.Builder()
            .setAdvertiseMode(AdvertiseSettings.ADVERTISE_MODE_LOW_LATENCY)
            .setTxPowerLevel(AdvertiseSettings.ADVERTISE_TX_POWER_HIGH)

            .setIncludeDeviceName(true)
            .addServiceUuid(ParcelUuid(serviceUuid))
            .addManufacturerData(0x1234, byteArrayOf(0x01, 0x02))
            .build()

        advertiser.startAdvertising(settings, data, advertiseCallback)
    }

    private val advertiseCallback = object : AdvertiseCallback() {
        override fun onStartSuccess(settingsInEffect: AdvertiseSettings) {
            Log.d("BLE", "Advertising started")
        }
        override fun onStartFailure(errorCode: Int) {
            Log.e("BLE", "Advertise failed: $errorCode")
        }
    }

                .setSsid("CarHotspot_5G")
                .setWpa2Passphrase("password123")
                .setIsAppInteractionRequired(false)
                .setPriority(10)
                .build(),

                .setSsid("CarHotspot_WPA3")

                .build(),

                .setPasspointConfig(buildPasspointConfig())
                .build()
        )

        val status = wifiManager.addNetworkSuggestions(suggestions)
        if (status != WifiManager.STATUS_NETWORK_SUGGESTIONS_SUCCESS) {
            Log.e("WiFi", "Suggestion failed: $status")
        }
    }

    // --- Wi-Fi Scan Results ---
    fun scanNetworks() {
        val receiver = object : BroadcastReceiver() {
            override fun onReceive(ctx: Context, intent: Intent) {
                val results = wifiManager.scanResults

                results.forEach { result ->
                    val ssid       = result.SSID
                    val bssid      = result.BSSID
                    val rssi       = result.level          // dBm
                    val freq       = result.frequency       // MHz
                    val security   = result.capabilities    // [WPA2][RSN]
                    val width      = result.channelWidth    // 20/40/80/160 MHz
                    val is6GHz    = freq > 5950
                    val is5GHz    = freq in 4900..5950
                }
            }
        }

        context.registerReceiver(receiver, IntentFilter(WifiManager.SCAN_RESULTS_AVAILABLE_ACTION))
        val started = wifiManager.startScan()
    }

    // --- ??????? ?????? ??????? ---

class WifiP2pManager(private val context: Context,
                       private val activity: Activity) {

    private val manager = context.getSystemService(WifiP2pManager::class.java)
    private val channel = manager.initialize(context, context.mainLooper, null)

    // --- Broadcast Receiver ??????? ---
    private val receiver = object : BroadcastReceiver() {
        override fun onReceive(ctx: Context, intent: Intent) {
            when (intent.action) {
                WifiP2pManager.WIFI_P2P_STATE_CHANGED_ACTION -> {
                    val state = intent.getIntExtra(WifiP2pManager.EXTRA_WIFI_STATE, -1)
                    val isEnabled = state == WifiP2pManager.WIFI_P2P_STATE_ENABLED
                }
                WifiP2pManager.WIFI_P2P_PEERS_CHANGED_ACTION -> {
                    // ????? ??????? ??????
                    manager.requestPeers(channel, peerListListener)
                }
                WifiP2pManager.WIFI_P2P_CONNECTION_CHANGED_ACTION -> {
                    val info: WifiP2pInfo? = intent.getParcelableExtra(
                        WifiP2pManager.EXTRA_WIFI_P2P_INFO
                    )
                    info?.let { handleConnection(it) }
                }
            }
        }
    }

    private val peerListListener = WifiP2pManager.PeerListListener { peers ->
        peers.deviceList.forEach { device ->
            Log.d("P2P", "Peer: ${device.deviceName} | ${device.deviceAddress}")
        }
    }

    fun discoverPeers() {
        manager.discoverPeers(channel, object : WifiP2pManager.ActionListener {
            override fun onSuccess() { Log.d("P2P", "Discovery started") }
            override fun onFailure(reason: Int) { Log.e("P2P", "Discovery failed: $reason") }
        })
    }

    fun connectToDevice(device: WifiP2pDevice) {
        val config = WifiP2pConfig().apply {
            deviceAddress = device.deviceAddress
            wps.setup = WpsInfo.PBC   // Push Button Connect
        }

        manager.connect(channel, config, object : WifiP2pManager.ActionListener {
            override fun onSuccess() { Log.d("P2P", "Connection initiated") }
            override fun onFailure(reason: Int) {}
        })
    }

                // ??? ??? Client Ś ??????? ???? GO

                Log.d("P2P", "Connect to GO at $goIp")
            }
        }
    }

class CarMusicService : MediaBrowserServiceCompat() {

    private lateinit var mediaSession: MediaSessionCompat

    override fun onCreate() {
        super.onCreate()

        mediaSession = MediaSessionCompat(this, "CarMusicService").apply {
            setFlags(MediaSessionCompat.FLAG_HANDLES_TRANSPORT_CONTROLS)
            setCallback(sessionCallback)
            isActive = true
        }

        sessionToken = mediaSession.sessionToken
    }

    // --- Media Callbacks (????? next/prev ?? ???????) ---
    private val sessionCallback = object : MediaSessionCompat.Callback() {

        override fun onPlay() {
            // ??? AudioFocus ?????
            val result = audioFocusRequest()
            if (result == AudioManager.AUDIOFOCUS_REQUEST_GRANTED) {
                startPlayback()
                mediaSession.isActive = true
                updatePlaybackState(PlaybackStateCompat.STATE_PLAYING)
            }
        }

        override fun onPause() {
            pausePlayback()
            updatePlaybackState(PlaybackStateCompat.STATE_PAUSED)
        }

        override fun onSkipToNext() { playNext() }
        override fun onSkipToPrevious() { playPrevious() }

        override fun onPlayFromMediaId(mediaId: String?, extras: Bundle?) {
            playTrack(mediaId ?: return)
        }
    }

    // --- ????? Metadata (??? ???????? ??????? ??????) ---
    fun updateMetadata(track: Track) {
        mediaSession.setMetadata(
            MediaMetadataCompat.Builder()
                .putString(MediaMetadataCompat.METADATA_KEY_TITLE, track.title)
                .putString(MediaMetadataCompat.METADATA_KEY_ARTIST, track.artist)
                .putString(MediaMetadataCompat.METADATA_KEY_ALBUM, track.album)
                .putBitmap(MediaMetadataCompat.METADATA_KEY_ALBUM_ART, track.artwork)
                .putLong(MediaMetadataCompat.METADATA_KEY_DURATION, track.durationMs)
                .build()
        )
    }

    // --- MediaBrowser Tree ---
    override fun onGetRoot(
        clientPackageName: String, clientUid: Int, rootHints: Bundle?
    ): BrowserRoot {
        // ???? ?? ?????? ?? Android Auto
        val isAndroidAuto = clientPackageName == "com.google.android.projection.gearhead"
        return BrowserRoot("root", null)
    }

    override fun onLoadChildren(
        parentId: String,
        result: Result<List<MediaBrowserCompat.MediaItem>>
    ) {
        val items = when (parentId) {
            "root" -> listOf(
                createBrowsableItem("playlists", "????? ???????"),
                createBrowsableItem("albums", "?????????"),
                createBrowsableItem("recent", "??????? ??????")
            )
