        }


    }

            enforceCarPermission()


            return carPropertyManager.getFloatProperty(


                VehiclePropertyIds.ENGINE_COOLANT_TEMP, 0


            )

            enforceCarPermission()

        override fun getFuelLevel(): Float {


            enforceCarPermission()


            return carPropertyManager.getFloatProperty(


                VehiclePropertyIds.FUEL_LEVEL, 0


            )


        }





        override fun getVehicleSnapshot(): VehicleData {


            enforceCarPermission()


            return VehicleData(


                speed   = getCurrentSpeed(),


                gear    = getCurrentGear(),


                fuel    = getFuelLevel(),


                temp    = getEngineTemp(),


                timestamp = System.currentTimeMillis()


            )

            enforceHvacPermission()


            val areaId = if (isDriverSide) VehicleAreaSeat.SEAT_ROW_1_LEFT


                         else VehicleAreaSeat.SEAT_ROW_1_RIGHT


            carPropertyManager.setFloatProperty(


                VehiclePropertyIds.HVAC_TEMPERATURE_SET,


                areaId, tempCelsius


            )


        }

            enforceHvacPermission()


            require(speed in 0..7) { "Fan speed must be 0-7" }


            carPropertyManager.setIntProperty(


                VehiclePropertyIds.HVAC_FAN_SPEED,


                VehicleAreaType.VEHICLE_AREA_TYPE_SEAT, speed


            )


        }





        override fun registerCallback(callback: ICarDataCallback) {


            enforceCarPermission()


            remoteCallbacks.register(callback)


        }





        override fun unregisterCallback(callback: ICarDataCallback) {


            remoteCallbacks.unregister(callback)

class VehicleMonitor(private val context: Context) {





    private var car: Car? = null


    private var propertyManager: CarPropertyManager? = null





    // ? ????? ?? Car Service


    fun connect() {


        car = Car.createCar(


            context,


            null,


            Car.CAR_WAIT_TIMEOUT_WAIT_FOREVER


        ) { newCar, ready ->


            if (ready) {


                propertyManager = newCar.getCarManager(Car.CAR_PROPERTY_SERVICE)


                    as CarPropertyManager


                startMonitoring()


            }


        }


    }





    // ---- Reading Properties ----





    fun getCurrentSpeed(): Float {


        return propertyManager?.getFloatProperty(


            VehiclePropertyIds.PERF_VEHICLE_SPEED,


            VehicleAreaType.VEHICLE_AREA_TYPE_GLOBAL  // areaId = 0


        ) ?: 0f


    }





    fun getOdometer(): Float {


        return propertyManager?.getFloatProperty(


            VehiclePropertyIds.PERF_ODOMETER, 0


        ) ?: 0f


    }





    fun isEngineRunning(): Boolean {


        val state = propertyManager?.getIntProperty(


            VehiclePropertyIds.ENGINE_IDLE_AUTO_STOP_ENABLED, 0


        )


        return state != null


    }





    // ---- Writing Properties ----





    fun setDriverSeatTemperature(level: Int) { // -3 to 3


        propertyManager?.setIntProperty(


            VehiclePropertyIds.HVAC_SEAT_TEMPERATURE,


            VehicleAreaSeat.SEAT_ROW_1_LEFT,


            level


        )


    }





    fun setInteriorLighting(brightness: Int) {


        propertyManager?.setIntProperty(


            VehiclePropertyIds.CABIN_LIGHTS_STATE, 0, brightness


        )


    }





    // ---- Continuous Monitoring (Callbacks) ----





    private val speedCallback = object : CarPropertyManager.CarPropertyEventCallback {





        override fun onChangeEvent(value: CarPropertyValue<*>) {


            when (value.propertyId) {


                VehiclePropertyIds.PERF_VEHICLE_SPEED -> {


                    val speed = value.value as Float


                    // CarPropertyValue.timestamp ?? nanoseconds


                    onSpeedUpdate(speed, value.timestamp)


                }


                VehiclePropertyIds.GEAR_SELECTION -> {


                    val gear = value.value as Int


                    onGearUpdate(gear)


                }


            }


        }





        override fun onErrorEvent(propId: Int, zone: Int) {


            Log.e("VehicleMonitor", "Error on propId: $propId, zone: $zone")


        }


    }





    private fun startMonitoring() {


        // UPDATE_RATE_FAST = ~10Hz, NORMAL = ~1Hz, UI = ~60Hz


        propertyManager?.registerCallback(


            speedCallback,


            VehiclePropertyIds.PERF_VEHICLE_SPEED,


            CarPropertyManager.SENSOR_RATE_FAST   // 10 updates/sec


        )


        propertyManager?.registerCallback(


            speedCallback,


            VehiclePropertyIds.GEAR_SELECTION,


            CarPropertyManager.SENSOR_RATE_NORMAL


        )


    }





    fun disconnect() {


        propertyManager?.unregisterCallback(speedCallback)


        car?.disconnect()


    }


}

class DashboardViewModel(application: Application) : AndroidViewModel(application) {





    private val _speed = MutableStateFlow(0f)


    val speed: StateFlow<Float> = _speed.asStateFlow()





    private val _gear = MutableStateFlow("P")

    private val _vehicleData = MutableStateFlow<VehicleData?>(null)


    val vehicleData = _vehicleData.asStateFlow()





    private var carService: ICarDataService? = null





    private val connection = object : ServiceConnection {


        override fun onServiceConnected(n: ComponentName, b: IBinder) {


            carService = ICarDataService.Stub.asInterface(b)


            carService?.registerCallback(callback)


            loadInitialState()


        }


        override fun onServiceDisconnected(n: ComponentName) {


            carService = null


            // ????? ??? OS ????? ??????? ????????


        }


    }





    private val callback = object : ICarDataCallback.Stub() {


        override fun onSpeedChanged(newSpeed: Float) {


            // StateFlow thread-safe — ???? ???????? ?? ?? thread


            _speed.value = newSpeed


        }


        override fun onGearChanged(old: String, new: String) {


            _gear.value = new


        }


        override fun onOverspeedWarning(speed: Float, limit: Float) { }


        override fun onError(code: Int, msg: String) { }


    }





    init { connectToCarService() }





    private fun connectToCarService() {


        val ctx = getApplication<Application>()


        ctx.bindService(


            Intent("com.example.car.ICarDataService"),


            connection,


            Context.BIND_AUTO_CREATE or Context.BIND_IMPORTANT

            }


        }


    }





    override fun onCleared() {


        carService?.unregisterCallback(callback)


        getApplication<Application>().unbindService(connection)


        super.onCleared()


    }


}
