NDK تفصيلي

Native Development Kit — كل تفصيلة فيه

📦 إيه هو NDK بالظبط؟

NDK هو مجموعة أدوات Google بتتيح لك تكتب أجزاء من تطبيق Android بـ C أو C++ بدل ما كل حاجة تكون Java/Kotlin. مش مجرد compiler، ده ecosystem كامل من الأدوات والمكتبات.

الـ compiler الأساسي. بيحول كود C/C++ لـ machine code مخصوص لمعالج ARM أو x86.

Java Native Interface — الجسر بين عالمين

🎯 مفهوم JNI

JNI هو specification بيحدد إزاي الكود Java/Kotlin ممكن يستدعي ويتم استدعاؤه من الكود Native (C/C++). هو البروتوكول — الاتفاقية — اللي بيتيح للعالمين يتكلموا.

تعقيد التعلم75%
القوة95%
خطر الـ Crash80%

🧩 JNIEnv — قلب JNI

هي struct بتحتوي على pointer لـ function table. كل thread عنده JNIEnv خاص بيه. من خلالها بتعمل كل العمليات.

C++
// JNIEnv* هي مدخلك لكل عمليات JNI
JNIEXPORT jstring JNICALL
Java_com_example_MainActivity_hello(
    JNIEnv* env,   // ← الجسر
    jobject thiz  // ← this (الـ object)
) {
    return env->NewStringUTF("Hello NDK!");
}

📛 قاعدة تسمية الـ Functions

لازم اسم الـ function في C++ يتطابق بالظبط مع مكان الـ method في Java/Kotlin:

C++ — تسمية صحيحة
// الصيغة: Java_[package]_[ClassName]_[methodName]
// Package: com.example.myapp → com_example_myapp
// ClassName: MainActivity
// MethodName: computeSum

JNIEXPORT jint JNICALL
Java_com_example_myapp_MainActivity_computeSum(
    JNIEnv* env,
    jobject thiz,
    jint a,
    jint b
) {
    return a + b;
}
⚠️
مهم: لو اسمك فيه underscore في package أو class name، لازم تحوله في JNI لـ _1 (يعني الـ underscore بتبقى _1 في الاسم).

⚡ Dynamic Registration — الطريقة الاحترافية

بدل ما تعتمد على اسم الـ function، تقدر تسجل الـ methods يدوياً في JNI_OnLoad — ده أسرع وأكثر مرونة:

C++ — JNI_OnLoad
#include <jni.h>
#include <string>

// الـ functions الفعلية
static jstring nativeHello(JNIEnv* env, jobject) {
    return env->NewStringUTF("Hello from C++!");
}

static jint nativeAdd(JNIEnv* env, jobject, jint a, jint b) {
    return a + b;
}

// جدول التسجيل
static const JNINativeMethod gMethods[] = {
    {"hello",  "()Ljava/lang/String;", (void*)nativeHello},
    {"add",    "(II)I",               (void*)nativeAdd},
};

// بيتنادى لما يتحمل الـ .so
jint JNI_OnLoad(JavaVM* vm, void*) {
    JNIEnv* env;
    vm->GetEnv((void**)&env, JNI_VERSION_1_6);

    jclass clazz = env->FindClass("com/example/MainActivity");
    env->RegisterNatives(clazz, gMethods, 2);

    return JNI_VERSION_1_6;
}

كتابة الـ C++ side

C++ — native-lib.cpp
#include <jni.h>
#include <string>
#include <android/log.h>

#define LOGI(...) \
  __android_log_print(ANDROID_LOG_INFO, "NDK", __VA_ARGS__)

extern "C" {

JNIEXPORT jstring JNICALL
Java_com_example_myapp_MainActivity_stringFromJNI(
    JNIEnv* env, jobject) {
    return env->NewStringUTF("أهلاً من C++!");
}

JNIEXPORT jint JNICALL
Java_com_example_myapp_MainActivity_addNumbers(
    JNIEnv* env, jobject, jint a, jint b) {
    LOGI("Adding %d + %d", a, b);
    return a + b;
}

JNIEXPORT jintArray JNICALL
Java_com_example_myapp_MainActivity_processArray(
    JNIEnv* env, jobject, jintArray input) {
    jsize len = env->GetArrayLength(input);
    jint* arr = env->GetIntArrayElements(input, nullptr);

    for (jsize i = 0; i < len; i++) arr[i] *= 2;

    env->ReleaseIntArrayElements(input, arr, 0);
    return input;
}

} // extern "C"

عمليات JNI المتقدمة

الـ Strings والـ Objects والـ Callbacks وإدارة الذاكرة

📝 التعامل مع الـ Strings

⚠️
String في Java ≠ String في C++. الـ jstring هي reference لـ Java object، مش char*. لازم تحولها وتحرر الذاكرة.
C++ — String Operations
// ① Java String → C++ string
JNIEXPORT jint JNICALL
Java_example_Cls_getLength(JNIEnv* env, jobject, jstring jstr) {
    const char* cstr = env->GetStringUTFChars(jstr, nullptr);
    int len = strlen(cstr);
    env->ReleaseStringUTFChars(jstr, cstr); // ← مهم جداً!
    return len;
}

// ② C++ string → Java String
JNIEXPORT jstring JNICALL
Java_example_Cls_getResult(JNIEnv* env, jobject) {
    std::string result = doSomeComputation();
    return env->NewStringUTF(result.c_str());
}

// ③ Unicode الصح — استخدم GetStringChars
jsize len = env->GetStringLength(jstr);
const jchar* chars = env->GetStringChars(jstr, nullptr);
// ... استخدام chars ...
env->ReleaseStringChars(jstr, chars);

🔄 Callbacks — Java من C++

تقدر تنادي methods في Java من كود C++ — مفيد جداً للـ async callbacks:

C++ — Calling Java Methods
void callJavaCallback(JNIEnv* env, jobject obj) {
    // ① جيب الـ class
    jclass clazz = env->GetObjectClass(obj);

    // ② جيب الـ method ID
    // Signature: "(I)V" = takes int, returns void
    jmethodID mid = env->GetMethodID(
        clazz, "onProgress", "(I)V"
    );

    if (mid == nullptr) return; // method مش موجودة

    // ③ نادي الـ method
    env->CallVoidMethod(obj, mid, 75); // 75%
}

// لناداء Static method:
jmethodID smid = env->GetStaticMethodID(
    clazz, "log", "(Ljava/lang/String;)V"
);
env->CallStaticVoidMethod(clazz, smid, env->NewStringUTF("done"));

الفصل السادس

إدارة الذاكرة في JNI

أخطر جزء — تجاهله = memory leaks وcrashes

❌ Local References

كل object بتنشئه في JNI function بيكون Local Reference — بتتحذف تلقائياً لما الـ function تخرج. لكن لو عندك loop كبيرة هتعمل memory overflow!

C++ — Local Ref Problem
// ❌ غلط — local refs بتتراكم
for (int i = 0; i < 10000; i++) {
    jstring s = env->NewStringUTF("hi");
    // ... بتعمل حاجة ...
    // بدون DeleteLocalRef → crash!
}

// ✅ صح
for (int i = 0; i < 10000; i++) {
    jstring s = env->NewStringUTF("hi");
    // ... بتعمل حاجة ...
    env->DeleteLocalRef(s); // ← ضروري!
}

✅ Global References

لو محتاج تحتفظ بـ reference بين calls مختلفة (زي caching الـ class ID)، استخدم Global Reference.

C++ — Global Ref
static jclass gMyClass = nullptr;

jint JNI_OnLoad(JavaVM* vm, void*) {
    JNIEnv* env;
    vm->GetEnv((void**)&env, JNI_VERSION_1_6);

    jclass local = env->FindClass("com/example/Cls");

    // تحويل local → global
    gMyClass = (jclass) env->NewGlobalRef(local);
    env->DeleteLocalRef(local);

    return JNI_VERSION_1_6;
}

void JNI_OnUnload(JavaVM* vm, void*) {
    JNIEnv* env;
    vm->GetEnv((void**)&env, JNI_VERSION_1_6);
    env->DeleteGlobalRef(gMyClass);
}

📋 قواعد إدارة الذاكرة — الملخص الذهبي

✅
دايماً ReleaseStringUTFChars بعد GetStringUTFChars
✅
دايماً ReleaseIntArrayElements بعد GetIntArrayElements
✅
استخدم DeleteLocalRef في loops طويلة
❌
متحتفظش بـ Local Ref بعد ما الـ JNI call تخلص
❌
متعملش DeleteGlobalRef على Local Reference
❌
متستخدمش JNIEnv في غير الـ thread اللي جتلك منه

تقنيات الاحتراف الحقيقي

⚡ Cache الـ IDs

FindClass و GetMethodID عمليات بطيئة — ابعدهم من الـ hot path:

C++
// ❌ بطيء — بيعمل lookup كل مرة
void processFrames(JNIEnv* env, ...) {
    jclass cls = env->FindClass("...");
    jmethodID mid = env->GetMethodID(...);
    // loop on every frame ← غلط!
}

// ✅ سريع — cache مرة واحدة
static jmethodID sCachedMid = nullptr;

jint JNI_OnLoad(JavaVM* vm, void*) {
    JNIEnv* env; ...
    jclass cls = env->FindClass("...");
    sCachedMid = env->GetMethodID(cls, ...);
    return JNI_VERSION_1_6;
}

🧮 Critical Sections

GetPrimitiveArrayCritical أسرع بكتير لكن بيوقف GC:

C++
void processPixels(JNIEnv* env, jbyteArray pixels) {
    // بيوقف GC مؤقتاً — أسرع بكتير!
    void* ptr =
        env->GetPrimitiveArrayCritical(pixels, nullptr);

    // ⚠️ هنا متعملش JNI calls تانية!
    applyFilter((uint8_t*)ptr, len);

    // لازم تحرر بسرعة
    env->ReleasePrimitiveArrayCritical(pixels, ptr, 0);
}

🧵 Threading في NDK

🚨
قاعدة ذهبية: JNIEnv مش thread-safe. كل thread لازم يعمل AttachCurrentThread قبل ما يستخدم JNI.
C++ — Safe Threading
static JavaVM* gJVM = nullptr;

jint JNI_OnLoad(JavaVM* vm, void*) {
    gJVM = vm; // احفظ الـ JavaVM
    return JNI_VERSION_1_6;
}

void workerThreadFunc() {
    JNIEnv* env = nullptr;

    // ① Attach الـ thread
    gJVM->AttachCurrentThread(&env, nullptr);

    // ② استخدم JNI بأمان
    jstring s = env->NewStringUTF("from thread");

    // ③ Detach قبل ما الـ thread تخلص
    gJVM->DetachCurrentThread();
}

💥 التعامل مع الـ Exceptions

C++ — Exception Handling
JNIEXPORT jint JNICALL
Java_example_Cls_riskyOp(JNIEnv* env, jobject, jint divisor) {

    // رمي exception من C++ إلى Java
    if (divisor == 0) {
        env->ThrowNew(
            env->FindClass("java/lang/ArithmeticException"),
            "Division by zero from NDK!"
        );
        return -1;
    }

    // التحقق من exception بعد JNI call
    someJNICall(env);
    if (env->ExceptionCheck()) {
        env->ExceptionDescribe(); // اطبع في logcat
        env->ExceptionClear();   // امسح العلامة
        return -1;
    }

    return 100 / divisor;
}

Checklist المحترف

قبل ما تعمل production release

✅ لازم تعمله

✓ cache الـ jclass و jmethodID في JNI_OnLoad
✓ Release كل String و Array بعد الاستخدام
✓ AttachCurrentThread في أي worker thread
✓ استخدم -std=c++17 و -O2 في الـ release
✓ فعّل ASan في debug builds للكشف عن memory issues

❌ ابعد عنه

✗ ماتستخدمش JNIEnv في غير الـ thread بتاعها
✗ ماتعملش JNI calls في Critical Section
✗ ماتنساش DeleteLocalRef في loops طويلة
✗ ماتعملش FindClass من worker thread بدون Global ClassLoader
✗ ماتنقلش JNIEnv* بين threads أبداً
