ConnectivityManager هو الـ API المركزي لإدارة الشبكة في Android. يسمح بمراقبة حالة الشبكة، طلب شبكات بمواصفات معينة، ومعرفة نوع الاتصال الحالي.

kotlin — ConnectivityManager الحديث (API 21+)
class NetworkMonitor(private val context: Context) {

    private val cm = context.getSystemService(ConnectivityManager::class.java)

    // ═══ الطريقة الحديثة: NetworkCallback ═══
    fun startMonitoring() {
        // تحديد نوع الشبكة المطلوبة
        val request = NetworkRequest.Builder()
            .addCapability(NetworkCapabilities.NET_CAPABILITY_INTERNET)
            .addCapability(NetworkCapabilities.NET_CAPABILITY_VALIDATED)
            .addTransportType(NetworkCapabilities.TRANSPORT_WIFI)
            .build()

        cm.registerNetworkCallback(request, networkCallback)
    }

    private val networkCallback = object : ConnectivityManager.NetworkCallback() {

        // شبكة جديدة متاحة
        override fun onAvailable(network: Network) {
            Log.d("Network", "Connected: $network")
        }

        // تغيرت خصائص الشبكة
        override fun onCapabilitiesChanged(
            network: Network,
            caps: NetworkCapabilities
        ) {
            val hasInternet = caps.hasCapability(NetworkCapabilities.NET_CAPABILITY_INTERNET)
            val isWifi      = caps.hasTransport(NetworkCapabilities.TRANSPORT_WIFI)
            val isCellular  = caps.hasTransport(NetworkCapabilities.TRANSPORT_CELLULAR)
            val bandwidth   = caps.linkDownstreamBandwidthKbps  // Kbps
            val signalStr   = caps.signalStrength  // -1 if unknown
        }

        // تغير في خصائص اللنك (IP, routes)
        override fun onLinkPropertiesChanged(network: Network, lp: LinkProperties) {
            val ip   = lp.linkAddresses
            val dns  = lp.dnsServers
            val iface = lp.interfaceName   // "wlan0", "rmnet0"
        }

        // الشبكة انقطعت
        override fun onLost(network: Network) {
            Log.d("Network", "Lost connection")
        }

        // الشبكة بطيئة جداً
        override fun onBlockedStatusChanged(network: Network, blocked: Boolean) {}
    }

    // ═══ تحديد شبكة معينة للطلبات ═══
    fun makeRequestOnSpecificNetwork(network: Network) {
        // ربط URL Connection بشبكة محددة
        val url = network.openConnection(java.net.URL("https://api.example.com"))
        // أو OkHttp:
        val socketFactory = network.socketFactory
        val client = OkHttpClient.Builder()
            .socketFactory(socketFactory).build()
    }

    fun stopMonitoring() = cm.unregisterNetworkCallback(networkCallback)
}

fun analyzeNetwork(caps: NetworkCapabilities) {
    // ── Transport Types ──
    val transports = buildString {
        if (caps.hasTransport(NetworkCapabilities.TRANSPORT_WIFI))     appendLine("WiFi")
        if (caps.hasTransport(NetworkCapabilities.TRANSPORT_CELLULAR)) appendLine("Cellular")
        if (caps.hasTransport(NetworkCapabilities.TRANSPORT_ETHERNET)) appendLine("Ethernet")
        if (caps.hasTransport(NetworkCapabilities.TRANSPORT_BLUETOOTH)) appendLine("Bluetooth")
        if (caps.hasTransport(NetworkCapabilities.TRANSPORT_USB))      appendLine("USB")
    }

    // ── Capabilities ──
    val isMetered  = !caps.hasCapability(NetworkCapabilities.NET_CAPABILITY_NOT_METERED)
    val isVPN      =  caps.hasCapability(NetworkCapabilities.NET_CAPABILITY_NOT_VPN).not()
    val isTrusted  =  caps.hasCapability(NetworkCapabilities.NET_CAPABILITY_TRUSTED)

    // ── Bandwidth estimation ──
    val downKbps = caps.linkDownstreamBandwidthKbps   // Kbps
    val upKbps   = caps.linkUpstreamBandwidthKbps

    // ── WiFi specific (API 29+) ──
    if (Build.VERSION.SDK_INT >= 29) {
        val wifiInfo = caps.transportInfo as? WifiInfo
        wifiInfo?.let {
            val ssid    = it.ssid          // اسم الشبكة
            val bssid   = it.bssid         // MAC للـ AP
            val rssi    = it.rssi           // قوة الإشارة dBm
            val freq    = it.frequency      // 2412=2.4GHz, 5180=5GHz
            val linkSpd = it.linkSpeed      // Mbps
        }
    }
}

kotlin — Bluetooth Discovery & Pairing
class BluetoothManager(private val context: Context) {

    private val adapter = BluetoothAdapter.getDefaultAdapter()
        ?: throw UnsupportedOperationException("No Bluetooth")

    // ═══ 1. تشغيل Bluetooth ═══
    fun enableBluetooth(activity: Activity) {
        if (!adapter.isEnabled) {
            // Android 12+: لا يمكن تشغيله برمجياً، يجب طلب المستخدم
            val intent = Intent(BluetoothAdapter.ACTION_REQUEST_ENABLE)
            activity.startActivityForResult(intent, REQ_ENABLE_BT)
        }
    }

    // ═══ 2. اكتشاف الأجهزة ═══
    private val discoveryReceiver = object : BroadcastReceiver() {
        override fun onReceive(ctx: Context, intent: Intent) {
            when (intent.action) {
                BluetoothDevice.ACTION_FOUND -> {
                    val device = intent.getParcelableExtra<BluetoothDevice>(
                        BluetoothDevice.EXTRA_DEVICE
                    )
                    val rssi = intent.getShortExtra(BluetoothDevice.EXTRA_RSSI, Short.MIN_VALUE)
                    device?.let {
                        Log.d("BT", "Found: ${it.name} | ${it.address} | RSSI: $rssi")
                    }
                }
                BluetoothAdapter.ACTION_DISCOVERY_FINISHED -> {
                    Log.d("BT", "Discovery done")
                }
            }
        }
    }

    fun startDiscovery() {
        val filter = IntentFilter().apply {
            addAction(BluetoothDevice.ACTION_FOUND)
            addAction(BluetoothAdapter.ACTION_DISCOVERY_FINISHED)
        }
        context.registerReceiver(discoveryReceiver, filter)
        adapter.startDiscovery()
    }

    // ═══ 3. رؤية الأجهزة المقترنة ═══
    fun getPairedDevices(): Set<BluetoothDevice> {
        return adapter.bondedDevices
    }

    // ═══ 4. اتصال SPP (Serial Port Profile) ═══
    fun connectToDevice(device: BluetoothDevice): BluetoothSocket {
        adapter.cancelDiscovery()   // مهم: يبطئ الاتصال
        val socket = device.createRfcommSocketToServiceRecord(
            UUID.fromString("00001101-0000-1000-8000-00805F9B34FB")
            // SPP UUID الرسمي
        )
        socket.connect()   // blocking — شغّله في background thread
        return socket
    }

    // ═══ 5. إرسال واستقبال بيانات ═══
    fun communicate(socket: BluetoothSocket) {
        val input  = socket.inputStream
        val output = socket.outputStream

        // إرسال
        output.write("AT+SPEED?\r\n".toByteArray())

        // استقبال (blocking — coroutine)
        val buffer = ByteArray(1024)
        val bytes = input.read(buffer)
        val message = String(buffer, 0, bytes)
    }
}

kotlin — A2DP & HFP Profile Management
class CarBluetoothProfiles(private val context: Context) {

    private var a2dpProfile: BluetoothA2dp? = null
    private var hfpProfile:  BluetoothHeadset? = null

    private val btAdapter = BluetoothAdapter.getDefaultAdapter()

    // ═══ الاتصال بـ Profile Proxy ═══
    fun initProfiles() {
        // A2DP — للصوت
        btAdapter.getProfileProxy(context, object : BluetoothProfile.ServiceListener {
            override fun onServiceConnected(profile: Int, proxy: BluetoothProfile) {
                a2dpProfile = proxy as BluetoothA2dp
            }
            override fun onServiceDisconnected(profile: Int) { a2dpProfile = null }
        }, BluetoothProfile.A2DP)

        // HFP — للمكالمات
        btAdapter.getProfileProxy(context, object : BluetoothProfile.ServiceListener {
            override fun onServiceConnected(profile: Int, proxy: BluetoothProfile) {
                hfpProfile = proxy as BluetoothHeadset
            }
            override fun onServiceDisconnected(profile: Int) { hfpProfile = null }
        }, BluetoothProfile.HEADSET)
    }

    // ═══ فحص حالة الـ Profile ═══
    fun getA2dpState(device: BluetoothDevice): Int {
        return a2dpProfile?.getConnectionState(device)
            ?: BluetoothProfile.STATE_DISCONNECTED
    }

    // ═══ مراقبة حالة A2DP ═══
    private val a2dpReceiver = object : BroadcastReceiver() {
        override fun onReceive(ctx: Context, intent: Intent) {
            val state = intent.getIntExtra(
                BluetoothProfile.EXTRA_STATE,
                BluetoothProfile.STATE_DISCONNECTED
            )
            when (state) {
                BluetoothProfile.STATE_CONNECTED    -> Log.d("A2DP", "Audio connected")
                BluetoothProfile.STATE_DISCONNECTED -> Log.d("A2DP", "Audio disconnected")
            }
        }
    }

    fun register() {
        context.registerReceiver(a2dpReceiver,
            IntentFilter(BluetoothA2dp.ACTION_CONNECTION_STATE_CHANGED))
    }
}

kotlin — BLE Scanning (Central Role)
class BleScanner(private val context: Context) {

    private val bleScanner = BluetoothAdapter.getDefaultAdapter()
        .bluetoothLeScanner

    private val scanCallback = object : ScanCallback() {

        override fun onScanResult(callbackType: Int, result: ScanResult) {
            val device    = result.device
            val rssi      = result.rssi           // قوة الإشارة
            val record    = result.scanRecord     // Advertisement data

            // اسم الجهاز من الـ Advertisement
            val localName = record?.deviceName

            // بيانات مخصصة من الشركة المصنّعة
            val mfrData   = record?.getManufacturerSpecificData(0x004C)  // Apple

            // Service UUIDs المُعلَن عنها
            val services  = record?.serviceUuids

            // اكتشاف نوع الجهاز
            val isCarKey = services?.any {
                it.uuid == UUID.fromString("0000FFF0-0000-1000-8000-00805F9B34FB")
            } == true
        }

        override fun onScanFailed(errorCode: Int) {
            Log.e("BLE", "Scan failed: $errorCode")
        }
    }

    fun startScan(targetServiceUuid: UUID? = null) {
        val filters = targetServiceUuid?.let { uuid ->
            listOf(ScanFilter.Builder()
                .setServiceUuid(ParcelUuid(uuid))
                .build())
        }

        val settings = ScanSettings.Builder()
            .setScanMode(ScanSettings.SCAN_MODE_LOW_LATENCY)   // أسرع لكن أكثر طاقة
            .setCallbackType(ScanSettings.CALLBACK_TYPE_ALL_MATCHES)
            .build()

        bleScanner.startScan(filters, settings, scanCallback)
    }

    fun stopScan() = bleScanner.stopScan(scanCallback)
}

class BleAdvertiser(private val context: Context) {

    private val advertiser = BluetoothAdapter.getDefaultAdapter()
        .bluetoothLeAdvertiser

    fun startAdvertising(serviceUuid: UUID) {
        val settings = AdvertiseSettings.Builder()
            .setAdvertiseMode(AdvertiseSettings.ADVERTISE_MODE_LOW_LATENCY)
            .setTxPowerLevel(AdvertiseSettings.ADVERTISE_TX_POWER_HIGH)
            .setConnectable(true)
            .build()

        val data = AdvertiseData.Builder()
            .setIncludeDeviceName(true)
            .addServiceUuid(ParcelUuid(serviceUuid))
            .addManufacturerData(0x1234, byteArrayOf(0x01, 0x02))
            .build()

        advertiser.startAdvertising(settings, data, advertiseCallback)
    }

    private val advertiseCallback = object : AdvertiseCallback() {
        override fun onStartSuccess(settingsInEffect: AdvertiseSettings) {
            Log.d("BLE", "Advertising started")
        }
        override fun onStartFailure(errorCode: Int) {
            Log.e("BLE", "Advertise failed: $errorCode")
        }
    }
}

class GattClient(private val context: Context) {

    companion object {
        val CAR_SERVICE_UUID     = UUID.fromString("0000FFF0-0000-1000-8000-00805F9B34FB")
        val SPEED_CHAR_UUID      = UUID.fromString("0000FFF1-0000-1000-8000-00805F9B34FB")
        val NOTIFY_CHAR_UUID     = UUID.fromString("0000FFF2-0000-1000-8000-00805F9B34FB")
        val CLIENT_CONFIG_UUID   = UUID.fromString("00002902-0000-1000-8000-00805F9B34FB")
    }

    private var gatt: BluetoothGatt? = null

    fun connect(device: BluetoothDevice) {
        gatt = device.connectGatt(context, false, gattCallback,
            BluetoothDevice.TRANSPORT_LE)
    }

    private val gattCallback = object : BluetoothGattCallback() {

        // 1. اتصال ناجح → اكتشف الـ Services
        override fun onConnectionStateChange(g: BluetoothGatt, status: Int, newState: Int) {
            if (newState == BluetoothProfile.STATE_CONNECTED) {
                Log.d("GATT", "Connected, discovering services...")
                g.discoverServices()
            }
        }

        // 2. Services اكتُشفت → اقرأ أو اشترك
        override fun onServicesDiscovered(g: BluetoothGatt, status: Int) {
            val service = g.getService(CAR_SERVICE_UUID) ?: return

            // قراءة Characteristic
            val speedChar = service.getCharacteristic(SPEED_CHAR_UUID)
            g.readCharacteristic(speedChar)

            // تفعيل Notifications
            val notifyChar = service.getCharacteristic(NOTIFY_CHAR_UUID)
            g.setCharacteristicNotification(notifyChar, true)

            // كتابة CCCD (Client Config Descriptor) لتفعيل notify
            val descriptor = notifyChar.getDescriptor(CLIENT_CONFIG_UUID)
            descriptor.value = BluetoothGattDescriptor.ENABLE_NOTIFICATION_VALUE
            g.writeDescriptor(descriptor)
        }

        // 3. استقبال قراءة
        override fun onCharacteristicRead(
            g: BluetoothGatt,
            char: BluetoothGattCharacteristic,
            status: Int
        ) {
            if (status == BluetoothGatt.GATT_SUCCESS) {
                val speed = char.getIntValue(BluetoothGattCharacteristic.FORMAT_UINT16, 0)
                Log.d("GATT", "Speed: $speed km/h")
            }
        }

        // 4. استقبال Notification
        override fun onCharacteristicChanged(
            g: BluetoothGatt,
            char: BluetoothGattCharacteristic
        ) {
            val data = char.value   // ByteArray
            // فك تشفير البيانات حسب بروتوكول الجهاز
            val speed = ((data[0].toInt() and 0xFF) or (data[1].toInt() shl 8))
        }
    }

    fun writeCommand(command: ByteArray) {
        val service = gatt?.getService(CAR_SERVICE_UUID) ?: return
        val char = service.getCharacteristic(SPEED_CHAR_UUID)
        char.value = command
        char.writeType = BluetoothGattCharacteristic.WRITE_TYPE_DEFAULT
        gatt?.writeCharacteristic(char)
    }

    fun disconnect() { gatt?.disconnect(); gatt?.close() }
}

class WifiConnector(private val context: Context) {

    private val wifiManager = context.getSystemService(WifiManager::class.java)

    // ═══ Network Suggestion (Android 10+) ═══
    fun suggestNetworks() {
        val suggestions = listOf(
            // WPA2 Network
            WifiNetworkSuggestion.Builder()
                .setSsid("CarHotspot_5G")
                .setWpa2Passphrase("password123")
                .setIsAppInteractionRequired(false)
                .setPriority(10)
                .build(),

            // WPA3 Network
            WifiNetworkSuggestion.Builder()
                .setSsid("CarHotspot_WPA3")
                .setWpa3Passphrase("SecurePass!456")
                .setIsMetered(false)
                .build(),

            // Enterprise / Passpoint
            WifiNetworkSuggestion.Builder()
                .setPasspointConfig(buildPasspointConfig())
                .build()
        )

        val status = wifiManager.addNetworkSuggestions(suggestions)
        if (status != WifiManager.STATUS_NETWORK_SUGGESTIONS_SUCCESS) {
            Log.e("WiFi", "Suggestion failed: $status")
        }
    }

    // ═══ Wi-Fi Scan Results ═══
    fun scanNetworks() {
        val receiver = object : BroadcastReceiver() {
            override fun onReceive(ctx: Context, intent: Intent) {
                val results = wifiManager.scanResults

                results.forEach { result ->
                    val ssid       = result.SSID
                    val bssid      = result.BSSID
                    val rssi       = result.level          // dBm
                    val freq       = result.frequency       // MHz
                    val security   = result.capabilities    // [WPA2][RSN]
                    val width      = result.channelWidth    // 20/40/80/160 MHz
                    val is6GHz    = freq > 5950
                    val is5GHz    = freq in 4900..5950
                }
            }
        }

        context.registerReceiver(receiver, IntentFilter(WifiManager.SCAN_RESULTS_AVAILABLE_ACTION))
        val started = wifiManager.startScan()
    }

    // ═══ معلومات الشبكة الحالية ═══
    fun getCurrentNetworkInfo(): WifiInfo {
        // API 29+: من NetworkCapabilities أفضل
        return wifiManager.connectionInfo.also { info ->
            Log.d("WiFi", "SSID: ${info.ssid}")
            Log.d("WiFi", "IP: ${info.ipAddress}")
            Log.d("WiFi", "RSSI: ${info.rssi} dBm")
            Log.d("WiFi", "Link Speed: ${info.linkSpeed} Mbps")
        }
    }
}

kotlin — Wi-Fi P2P Discovery & Connection
class WifiP2pManager(private val context: Context,
                       private val activity: Activity) {

    private val manager = context.getSystemService(WifiP2pManager::class.java)
    private val channel = manager.initialize(context, context.mainLooper, null)

    // ═══ Broadcast Receiver للأحداث ═══
    private val receiver = object : BroadcastReceiver() {
        override fun onReceive(ctx: Context, intent: Intent) {
            when (intent.action) {
                WifiP2pManager.WIFI_P2P_STATE_CHANGED_ACTION -> {
                    val state = intent.getIntExtra(WifiP2pManager.EXTRA_WIFI_STATE, -1)
                    val isEnabled = state == WifiP2pManager.WIFI_P2P_STATE_ENABLED
                }
                WifiP2pManager.WIFI_P2P_PEERS_CHANGED_ACTION -> {
                    // قائمة الأجهزة تغيّرت
                    manager.requestPeers(channel, peerListListener)
                }
                WifiP2pManager.WIFI_P2P_CONNECTION_CHANGED_ACTION -> {
                    val info: WifiP2pInfo? = intent.getParcelableExtra(
                        WifiP2pManager.EXTRA_WIFI_P2P_INFO
                    )
                    info?.let { handleConnection(it) }
                }
            }
        }
    }

    private val peerListListener = WifiP2pManager.PeerListListener { peers ->
        peers.deviceList.forEach { device ->
            Log.d("P2P", "Peer: ${device.deviceName} | ${device.deviceAddress}")
        }
    }

    fun discoverPeers() {
        manager.discoverPeers(channel, object : WifiP2pManager.ActionListener {
            override fun onSuccess() { Log.d("P2P", "Discovery started") }
            override fun onFailure(reason: Int) { Log.e("P2P", "Discovery failed: $reason") }
        })
    }

    fun connectToDevice(device: WifiP2pDevice) {
        val config = WifiP2pConfig().apply {
            deviceAddress = device.deviceAddress
            wps.setup = WpsInfo.PBC   // Push Button Connect
        }

        manager.connect(channel, config, object : WifiP2pManager.ActionListener {
            override fun onSuccess() { Log.d("P2P", "Connection initiated") }
            override fun onFailure(reason: Int) {}
        })
    }

    private fun handleConnection(info: WifiP2pInfo) {
        if (info.groupFormed) {
            if (info.isGroupOwner) {
                // أنا الـ Group Owner (Server)
                Log.d("P2P", "I'm the GO. Listen on socket.")
                // ServerSocket يستقبل connections
            } else {
                // أنا الـ Client — الاتصال بالـ GO
                val goIp = info.groupOwnerAddress.hostAddress
                Log.d("P2P", "Connect to GO at $goIp")
            }
        }
    }
}

Wireless Android Auto
البروتوكول الكامل — من Discovery للـ Video Stream
1

Bluetooth Discovery

السيارة تُرسل Bluetooth advertisement. الهاتف يكتشفها عبر BLE scan. الاتصال الأولي يتم عبر Bluetooth لتبادل معلومات الـ Wi-Fi.

2

Wi-Fi Credentials Exchange

السيارة تُرسل SSID وكلمة مرور الـ Wi-Fi Hotspot عبر Bluetooth. الهاتف يتصل بـ hotspot السيارة تلقائياً.

3

TCP Connection

الهاتف يفتح TCP connection للسيارة على منفذ محدد. يبدأ بروتوكول Android Auto الكامل عبر هذا الـ socket.

4

Video Streaming

الهاتف يُرسل H.264/H.265 encoded video stream. السيارة تُفكّك الضغط وتعرض على شاشتها. latency: ~100-200ms.

5

Audio & Input

صوت يُرسل كـ PCM (أو AAC) عبر نفس الـ connection. لمس الشاشة يُرسل كـ touch events للهاتف. Microphone يُرسل للهاتف أيضاً.

// MediaBrowserServiceCompat — للـ Media Apps في Android Auto
class CarMusicService : MediaBrowserServiceCompat() {

    private lateinit var mediaSession: MediaSessionCompat

    override fun onCreate() {
        super.onCreate()

        mediaSession = MediaSessionCompat(this, "CarMusicService").apply {
            setFlags(MediaSessionCompat.FLAG_HANDLES_TRANSPORT_CONTROLS)
            setCallback(sessionCallback)
            isActive = true
        }

        sessionToken = mediaSession.sessionToken
    }

    // ═══ Media Callbacks (أزرار next/prev في السيارة) ═══
    private val sessionCallback = object : MediaSessionCompat.Callback() {

        override fun onPlay() {
            // طلب AudioFocus أولاً
            val result = audioFocusRequest()
            if (result == AudioManager.AUDIOFOCUS_REQUEST_GRANTED) {
                startPlayback()
                mediaSession.isActive = true
                updatePlaybackState(PlaybackStateCompat.STATE_PLAYING)
            }
        }

        override fun onPause() {
            pausePlayback()
            updatePlaybackState(PlaybackStateCompat.STATE_PAUSED)
        }

        override fun onSkipToNext() { playNext() }
        override fun onSkipToPrevious() { playPrevious() }

        override fun onPlayFromMediaId(mediaId: String?, extras: Bundle?) {
            playTrack(mediaId ?: return)
        }
    }

    // ═══ تحديث Metadata (اسم الأغنية، الفنان، الغلاف) ═══
    fun updateMetadata(track: Track) {
        mediaSession.setMetadata(
            MediaMetadataCompat.Builder()
                .putString(MediaMetadataCompat.METADATA_KEY_TITLE, track.title)
                .putString(MediaMetadataCompat.METADATA_KEY_ARTIST, track.artist)
                .putString(MediaMetadataCompat.METADATA_KEY_ALBUM, track.album)
                .putBitmap(MediaMetadataCompat.METADATA_KEY_ALBUM_ART, track.artwork)
                .putLong(MediaMetadataCompat.METADATA_KEY_DURATION, track.durationMs)
                .build()
        )
    }

    // ═══ MediaBrowser Tree ═══
    override fun onGetRoot(
        clientPackageName: String, clientUid: Int, rootHints: Bundle?
    ): BrowserRoot {
        // تحقق أن المتصل هو Android Auto
        val isAndroidAuto = clientPackageName == "com.google.android.projection.gearhead"
        return BrowserRoot("root", null)
    }

    override fun onLoadChildren(
        parentId: String,
        result: Result<List<MediaBrowserCompat.MediaItem>>
    ) {
        val items = when (parentId) {
            "root" -> listOf(
                createBrowsableItem("playlists", "قوائم التشغيل"),
                createBrowsableItem("albums", "الألبومات"),
                createBrowsableItem("recent", "المشغّل مؤخراً")
            )
            "recent" -> getRecentTracks().map { createPlayableItem(it) }
            else -> emptyList()
        }
        result.sendResult(items)
    }
}

class AaosBluetoothManager(private val context: Context) {

    private val car = Car.createCar(context)
    private val btManager = car.getCarManager(Car.BLUETOOTH_SERVICE)
        as CarBluetoothManager

    // الأجهزة المقترنة مع السيارة
    fun getCarBluetoothDevices(): List<BluetoothDevice> {
        return btManager.getProfileDevicePriorityList(BluetoothProfile.A2DP)
    }

    // تعيين أولوية جهاز معين
    fun setDevicePriority(device: BluetoothDevice, priority: Int) {
        btManager.setProfileDevicePriority(device, BluetoothProfile.A2DP, priority)
    }
}

            }

    }

أخطاء شائعة وأدوات Debug
المشاكل التي ستواجهها وكيف تحلها
❌ BLE Scan لا يجد أجهزة

الأسباب: Location Permission مرفوضة، Location Services متوقفة، الجهاز لا يدعم BLE. الحل: تحقق من BluetoothAdapter.isLeEnabled() وfallback لـ Classic scan.

❌ GATT Connection يتقطع

السبب: GATT connection timeout (رأس المال: 30 ثانية بدون بيانات). الحل: أرسل GATT_WRITE_CHARACTERISTIC كـ Keep-Alive كل 15 ثانية.

❌ Android Auto لا يتصل Wireless

الأسباب: الهاتف ليس Android 11+، البيئة 2.4GHz (يحتاج 5GHz)، جدار الحماية يمنع TCP port. الحل: تأكد من Wi-Fi 5GHz وفتح المنافذ.

❌ Car App Template Error

السبب: تجاوز حد الـ Items (10 rows في ListTemplate). الحل: استخدم pagination أو SectionedItemList مع حد 6 items لكل قسم.
