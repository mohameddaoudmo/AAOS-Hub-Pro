📖 مثال شامل: Two Sum في Sorted Array

📝

المسألة: array مرتب، إيجاد اتنين عناصر مجموعهم = target

الحل: L من اليسار، R من اليمين. لو المجموع أكبر من target نقلل R، لو أصغر نزود L

vector<int> twoSum(vector<int>& arr, int target) {




    int L = 0, R = arr.size() - 1;




    while (L < R) {




        int curr = arr[L] + arr[R];




        if (curr == target) {




            return {L, R};




        } else if (curr < target) {




            L++;




        } else {




            R--;




        }




    }




    return {};




}




// Time: O(n) | Space: O(1)

📖 مثال: Container With Most Water

📝

إيجاد أكبر كمية مياه بين عمودين

L من اليسار، R من اليمين
نحسب المساحة = min(h[L], h[R]) × (R - L)
نحرك المؤشر الأقصر → لأنه هو اللي بيحدد الارتفاع
نكرر حتى يلتقوا
int maxWater(vector<int>& heights) {




    int L = 0, R = heights.size() - 1;




    int maxW = 0;




    while (L < R) {




        int h = min(heights[L], heights[R]);




        maxW = max(maxW, h * (R - L));




        if (heights[L] < heights[R]) {




            L++;




        } else {




            R--;




        }




    }




    return maxW;




}




// Time: O(n) | Space: O(1)

📖 مثال: 3Sum = Three Pointers!

vector<vector<int>> threeSum(vector<int>& nums) {




    sort(nums.begin(), nums.end());




    vector<vector<int>> result;




    for (int i = 0; i < (int)nums.size() - 2; ++i) {




        if (i > 0 && nums[i] == nums[i-1]) continue;




        int L = i + 1, R = nums.size() - 1;




        while (L < R) {




            int sum = nums[i] + nums[L] + nums[R];




            if (sum == 0) {




                result.push_back({nums[i], nums[L], nums[R]});




                while (L < R && nums[L] == nums[L+1]) L++;




                while (L < R && nums[R] == nums[R-1]) R--;




                L++; R--;




            } else if (sum < 0) L++;




            else R--;




        }




    }




    return result;




}




// Time: O(n²) | Space: O(1)

Sliding Window — النافذة المتحركة

الـ pattern الأقوى للـ subarrays والـ substrings

💡

الفكرة: بدل ما نحسب كل subarray من الصفر، بنحرك "نافذة" على الـ array ونحدّث الحساب تدريجياً. من O(n²) أو O(n³) → O(n)

int maxSumFixed(vector<int>& arr, int k) {




    // First window




    int curr = 0;




    for (int i = 0; i < k; ++i) curr += arr[i];




    




    int best = curr;




    for (int i = k; i < arr.size(); ++i) {




        // Add new, remove old




        curr += arr[i] - arr[i - k];




        best = max(best, curr);




    }




    return best;




}




// Time: O(n) | Space: O(1)

🏗️ Template عام للـ Sliding Window

int slidingWindow(vector<int>& arr) {




    int L = 0, result = 0;




    unordered_map<int, int> window_state;









    for (int R = 0; R < arr.size(); ++R) {




        // 1. ADD arr[R] to window




        window_state[arr[R]]++;









        // 2. SHRINK if invalid




        while (/* window invalid */) {




            window_state[arr[L]]--;




            if (window_state[arr[L]] == 0) {




                window_state.erase(arr[L]);




            }




            L++;




        }









        // 3. UPDATE result with valid window [L..R]




        result = max(result, R - L + 1);




    }




    return result;




}

📖 Longest Substring Without Repeating Characters

📝

أطول substring مش فيها حروف متكررة

R يمشي على كل حرف
لو الحرف موجود في الـ window، حرك L لبعده
حدّث max length
"abcabcbb"
window "abc" → len=3
when 'a' repeats → L moves
int lengthOfLongest(string s) {




    int L = 0, best = 0;




    unordered_map<char, int> seen; // char -> last index









    for (int R = 0; R < s.length(); ++R) {




        if (seen.count(s[R])) {




            // Move L past last occurrence




            L = max(L, seen[s[R]] + 1);




        }




        seen[s[R]] = R;




        best = max(best, R - L + 1);




    }




    return best;




}




// Time: O(n) | Space: O(min(n,k))

📖 Minimum Window Substring (Hard)

string minWindow(string s, string t) {




    unordered_map<char, int> need, window;




    for (char c : t) need[c]++;




    




    int have = 0, total = need.size();




    int resL = -1, minLen = INT_MAX, L = 0;









    for (int R = 0; R < s.length(); ++R) {




        char c = s[R];




        if (need.count(c)) {




            window[c]++;




            if (window[c] == need[c]) have++;




        }









        while (have == total) {




            if (R - L + 1 < minLen) {




                minLen = R - L + 1;




                resL = L;




            }




            char leftChar = s[L];




            if (need.count(leftChar)) {




                if (window[leftChar] == need[leftChar]) have--;




                window[leftChar]--;




            }




            L++;




        }




    }




    return minLen == INT_MAX ? "" : s.substr(resL, minLen);




}




// Time: O(n+m) | Space: O(n+m)

Prefix Sums — المجاميع التراكمية

Preprocessing technique تحول O(n) per query → O(1)

💡

الفكرة: نبني array جديد prefix بحيث prefix[i] = sum(arr[0..i]). بعدين أي range sum بيتحسب في O(1) بـ: prefix[r] - prefix[l-1]

🔧 بناء الـ Prefix Sum Array

vector<int> buildPrefix(vector<int>& arr) {




    int n = arr.size();




    vector<int> prefix(n + 1, 0);  // +1 trick!




    for (int i = 0; i < n; ++i) {




        prefix[i + 1] = prefix[i] + arr[i];




    }




    return prefix;




}









int rangeSum(vector<int>& prefix, int l, int r) {




    // sum of arr[l..r] inclusive




    return prefix[r + 1] - prefix[l];




}




// Build: O(n) | Query: O(1)

🗺️ 2D Prefix Sum

لـ matrices! نفس الفكرة بس في بعدين

// Build 2D prefix




for (int i = 1; i <= m; ++i) {




    for (int j = 1; j <= n; ++j) {




        P[i][j] = P[i-1][j] + P[i][j-1]




                - P[i-1][j-1] + grid[i-1][j-1];




    }




}









// Query: sum of rectangle (r1,c1) to (r2,c2)




int query(int r1, int c1, int r2, int c2) {




    return P[r2+1][c2+1] - P[r1][c2+1]




         - P[r2+1][c1] + P[r1][c1];




}

📖 Subarray Sum Equals K — الأقوى!

📝

عدد الـ subarrays اللي مجموعها = k

نبني prefix sum وهنا بنحتاج HashMap
لو prefix[r] - prefix[l] = k → يعني prefix[l] = prefix[r] - k
نبحث في الـ HashMap عن curr - k
هذا بيحول O(n²) → O(n)!
int subarraySum(vector<int>& nums, int k) {




    int count = 0, curr = 0;




    unordered_map<int, int> seen;




    seen[0] = 1; // important!









    for (int num : nums) {




        curr += num;




        if (seen.count(curr - k)) {




            count += seen[curr - k];




        }




        seen[curr]++;




    }




    return count;




}




// Time: O(n) | Space: O(n)

📖 Find Pivot Index

int pivotIndex(vector<int>& nums) {




    int total = accumulate(nums.begin(), nums.end(), 0);




    int left = 0;




    for (int i = 0; i < nums.size(); ++i) {




        // right sum = total - left - nums[i]




        if (left == total - left - nums[i]) {




            return i;




        }




        left += nums[i];




    }




    return -1;




}




// Time: O(n) | Space: O(1)

  • 🔑 الـ array أو string مرتب (sorted)

  • 🔑 "in-place reversal" أو palindrome
  • 🔑 "remove duplicates from sorted array"
  • 🔑 "merge two sorted arrays"

  • 🔑 "two numbers" مع sorted condition
  • 🔑 Cycle detection في linked list

🪟 إشارات Sliding Window

  • 🔑 "longest/shortest subarray/substring that..."
  • 🔑 "subarray with max/min sum of size k"
  • 🔑 "at most k distinct characters"
  • 🔑 "contiguous" أو "consecutive"
  • 🔑 كلام عن window أو subrange
  • 🔑 String أو array + condition على elements
  • 🔑 "no repeating characters"
  • 🔑 "contains all characters of t"

➕ إشارات Prefix Sums

  • 🔑 "range sum queries" (أكثر من query)
  • 🔑 "number of subarrays with sum = k"
  • 🔑 "subarray sum divisible by k"
  • 🔑 "find subarray with equal 0s and 1s"
  • 🔑 Matrix rectangle sum queries
  • 🔑 "count subarrays where condition on sum"
  • 🔑 الـ array فيه negative numbers + sum queries
  • 🔑 "running total" أو "cumulative"

🤔 امتى تستخدم HashMap مع Prefix?

  • 🔑 "count subarrays with sum = k" (مش max أو min)
  • 🔑 عايز تبحث عن prefix[r] - k
  • 🔑 Subarray sum divisible by k (modulo trick)
  • 🔑 Longest subarray with equal 0s & 1s
  • 🔑 "balanced" subarray problems
  • 🔑 أي وقت تحتاج تبحث عن prefix سابق بسرعة
